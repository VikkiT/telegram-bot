import telebot
import os

TOKEN = os.getenv("BOT_TOKEN")  # безопасно получаем токен из переменной окружени€
CHANNELS = os.getenv("CHANNELS").split(',')  # список каналов через зап€тую

bot = telebot.TeleBot(TOKEN)

@bot.message_handler(commands=['send'])
def send_message(message):
    if message.chat.type == 'private':
        text = message.text[6:]
        for ch in CHANNELS:
            try:
                bot.send_message(ch.strip(), text)
            except Exception as e:
                bot.send_message(message.chat.id, f'ќшибка дл€ {ch}: {e}')
        bot.send_message(message.chat.id, "–ассылка завершена.")

bot.polling()
